树状数组作用：在log时间里进行单点修改并维护前缀和
离散化求逆序对

int d[N+5];
int lowbit(int x)
{
    return x&-x;
}

int query(int p)//查询p的前缀和
{
    int ans=0;
    while(p)
    {
        ans+=d[p];
        p-=lowbit(p);
    }
    return ans;
}

void update(int p,int val)//p点更新
{
    while(p<1e6)
    {
        d[p]+=val;
        p+=lowbit(p);
    }
}
